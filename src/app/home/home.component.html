Setup

Follow these simple steps to get started

1 Installation
2 Register RoutingModule
3 Add to feature modules
4 Add routes 
npm i --save angular-routing
ng add @angular/material
import {{ '{' }} RoutingModule } from 'angular-routing';

@NgModule({{ '{' }}
    imports: [
    // ... other imports
    RoutingModule.forRoot()
  ]
})
export class AppModule {{ '{' }}}
import {{ '{' }} RoutingModule } from 'angular-routing';

@NgModule({{ '{' }}
    imports: [
    // ... other imports
    RoutingModule
  ]
})
export class FeatureModule {{ '{' }}}
<router>
  <route path="/blog/**">
    <app-blog *routeComponent></app-blog>
  </route>
  <route path="/posts/:postId">
    <app-post *routeComponent></app-post>
  </route>
  <route path="/about">
    <app-about *routeComponent></app-about>
  </route>
  <route path="/" redirectTo="/blog">
  </route>
  <route path="**">
    <app-page-not-found *routeComponent></app-page-not-found>
  </route>
</router>
